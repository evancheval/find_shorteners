xan select links ../postbayrou.csv | xan filter 'links' | xan explode links | xan freq -s links -A | xan select value,count | xan rename link -s value > links.csv
minet resolve link -i links.csv > resolved_links.csv
minet url-parse link -i resolved_links.csv | xan filter "probably_shortened eq 'no'" | xan select link,count,resolved_url,domain_name | xan rename original_domain -s domain_name | xan filter 'original_domain' > original_domains.csv
minet url-parse resolved_url -i original_domains.csv | xan select link,count,resolved_url,domain_name,original_domain | xan rename resolved_domain -s domain_name | xan filter 'resolved_domain' > domains.csv
xan filter 'original_domain ne resolved_domain' domains.csv | xan select link,original_domain,resolved_domain > probably_shortened_not_already_in_ural.csv
